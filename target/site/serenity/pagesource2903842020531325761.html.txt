            function hideAllPopOvers(objectName) {
                $('.slds-dropdown-trigger.slds-dropdown-trigger_click').each(function (index, value) {
                    //all buttons different than me (sObjectid)
                    if (!value.id.endsWith(objectName)) {
                        if (value.id) {
                            $('#' + value.id).removeClass('slds-is-open');
                        }
                    }
                });

                $('section.slds-popover').each(function (index, value) {
                    if (!value.id.endsWith(objectName)) {
                        $(this).removeClass('slds-visble');
                        $(this).addClass('slds-hidden');
                    }
                });
            }

            function toggleActionMenu(objectName) {
                //close others dropdowns
                hideAllPopOvers(objectName);
                $('#actionMenu_' + objectName).toggleClass('slds-is-open');
            }

            function toggleUserPopOver(sObjectid = '') {
                //close others dropdowns
                hideAllPopOvers(sObjectid);

                $('#actionMenu_' + sObjectid).toggleClass('slds-visible');
                $('#actionMenu_' + sObjectid).toggleClass('slds-hidden');
            }
